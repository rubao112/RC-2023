58// Determine the current state of the state machine
59void stateDetermine(stateMachine *state, char byte, int user)
60{
61 unsigned char FLAG_A = 0x03; // Address flag constant
62 unsigned char FLAG_C = (user) ? 0x07 : 0x03; // Control flag constant, varies based on user role (transmitter or receiver)
63
64 // Switch through the different states of the state machine
65 switch (*state)
66 {
67 case START:
68 // If FLAG is received, transition to the FLAG_RCV state
69 if (byte == FLAG)
70 *state = FLAG_RCV;
71 break;
72
73 case FLAG_RCV:
74 // If another FLAG is received, stay in FLAG_RCV state
75 if (byte == FLAG)
76 *state = FLAG_RCV;
77 // If FLAG_A (address byte) is received, transition to the A_RCV state
78 else if (byte == FLAG_A)
79 *state = A_RCV;
80 // Any other byte, revert back to the START state
81 else
82 *state = START;
83 break;
84
85 case A_RCV:
86 // If FLAG is received, transition to the FLAG_RCV state
87 if (byte == FLAG)
88 *state = FLAG_RCV;
89 // If FLAG_C (control byte) is received, transition to the C_RCV state
90 else if (byte == FLAG_C)
91 *state = C_RCV;
92 // Any other byte, revert back to the START state
93 else
94 *state = START;
95 break;
96
97 case C_RCV:
98 // If FLAG is received, transition to the FLAG_RCV state
99 if (byte == FLAG)
100 *state = FLAG_RCV;
101 // If BCC (checksum of address and control bytes) is received, transition to the BCC_NORMAL state
102 else if (byte == BCC(FLAG_A, FLAG_C))
103 *state = BCC_NORMAL;
104 // Any other byte, revert back to the START state
105 else
106 *state = START;
107 break;
108
109 case BCC_NORMAL:
110 // If FLAG is received, marking the end of the frame, transition to the DONE state
111 if (byte == FLAG)
112 *state = DONE;
113 // Any other byte, revert back to the START state
114 else
115 *state = START;
116 break;
117
118 default:
119 break;
120 }
121}

272// Process received ACK messages
273void receiveACK(stateMachine *state, unsigned char byte, unsigned char *ack, int sequenceNum)
274{
275 switch (*state)
276 {
277 case START:
278 // If the byte is a FLAG, transition to the FLAG_RCV state.
279 if (byte == FLAG)
280 {
281 *state = FLAG_RCV;
282 }
283 break;
284
285 case FLAG_RCV:
286 // On receiving a FLAG, remain in the FLAG_RCV state.
287 if (byte == FLAG)
288 {
289 *state = FLAG_RCV;
290 }
291 // On receiving A, transition to the A_RCV state.
292 else if (byte == A)
293 {
294 *state = A_RCV;
295 }
296 // Any other byte, revert to the START state.
297 else
298 {
299 *state = START;
300 }
301 break;
302
303 case A_RCV:
304 // If a FLAG is received, return to the FLAG_RCV state.
305 if (byte == FLAG)
306 {
307 *state = FLAG_RCV;
308 }
309 // If the byte matches the expected ACK, transition to the C_RCV state.
310 else if (byte == ACK(sequenceNum))
311 {
312 *state = C_RCV;
313 *ack = ACK(sequenceNum);
314 }
315 // If the byte matches the expected NACK, also transition to the C_RCV state.
316 else if (byte == NACK(sequenceNum))
317 {
318 *state = C_RCV;
319 *ack = NACK(sequenceNum);
320 }
321 // Any other byte, revert to the START state.
322 else
323 {
324 *state = START;
325 }
326 break;
327
328 case C_RCV:
329 // If the byte matches the expected BCC value, transition to the BCC_NORMAL state.
330 if (byte == BCC(A, *ack))
331 {
332 *state = BCC_NORMAL;
333 }
334 // If a FLAG is received, reset the acknowledgment and return to the FLAG_RCV state.
335 else if (byte == FLAG_RCV)
336 {
337 *ack = FALSE;
338 *state = FLAG_RCV;
339 }
340 // For any other byte, reset acknowledgment and revert to the START state.
341 else
342 {
343 *state = START;
344 *ack = FALSE;
345 }
346 break;
347
348 case BCC_NORMAL:
349 // If a FLAG is received after BCC, transition to the DONE state, indicating a complete ACK/NACK frame.
350 if (byte == FLAG)
351 {
352 *state = DONE;
353 }
354 // Any other byte, reset the acknowledgment and revert to the START state.
355 else
356 {
357 *ack = FALSE;
358 *state = START;
359 }
360 break;
361
362 default:
363 break;
364 }
365}

367// Sends a data buffer over the link layer, handles byte stuffing, and acknowledges receipt.
368int llwrite(const unsigned char *buf, int bufSize)
369{
370 signal(SIGALRM, alarmManager); // Register the alarm signal manager
371 state = START;
372 int attemptNum = 0; // Counter for retry attempts
373
374 // Count bytes that need stuffing
375 unsigned int counter = 0;
376 for (int i = 0; i < bufSize; i++)
377 {
378 if (buf[i] == FLAG || buf[i] == ESC)
379 {
380 counter++;
381 }
382 }
383
384 unsigned int size = bufSize + 6 + counter; // Calculate total size after stuffing
385 unsigned char message[size];
386 memset(message, 0, size); // Initialize the message buffer
387
388 // Frame header
389 message[0] = FLAG;
390 message[1] = A;
391 message[2] = sequenceNum << 7;
392 message[3] = BCC(A, sequenceNum << 7);
393
394 unsigned int BCC2 = 0, i = 0;
395
396 // Byte stuffing for the message
397 for (int j = 0; j < bufSize; j++)
398 {
399 switch (buf[j])
400 {
401 case ESC:
402 message[i + j + 4] = ESC;
403 message[i + j + 5] = ESC ^0x20;
404 BCC2 = BCC(BCC2, ESC);
405 i++;
406 break;
407
408 case FLAG:
409 message[i + j + 4] = ESC;
410 message[i + j + 5] = FLAG ^0x20;
411 BCC2 = BCC(BCC2, FLAG);
412 i++;
413 break;
414
415 default:
416 message[i + j + 4] = buf[j];
417 BCC2 = BCC(BCC2, buf[j]);
418 }
419 }
420
421 message[i + bufSize + 4] = BCC2; // Frame footer with BCC and FLAG
422 message[i + bufSize + 5] = FLAG;
423
424 STOP = FALSE;
425 alarmOn = FALSE;
426
427 // Loop until the frame is acknowledged or the maximum number of attempts is reached
428 while (STOP != TRUE && state != DONE)
429 {
430 unsigned char ack; // ACK byte
431 unsigned char receivedByte; // Byte read from the link
432
433 // Send the frame if the alarm is not already set
434 if (alarmOn == FALSE)
435 {
436 if (attemptNum > linkLayer.nRetransmissions)
437 {
438 return -1; // Max attempts reached
439 }
440 attemptNum++;
441
442 write(fd, message, size); // Send the message
443 signal(SIGALRM, alarmManager); // Set the alarm signal manager again
444 alarm(3); // Set an alarm for 3 seconds
445 alarmOn = TRUE;
446 }
447
448 // Read a byte from the link
449 unsigned int bytesNum = read(fd, &receivedByte, 1);
450
451 if (bytesNum > 0)
452 {
453 receiveACK(&state, receivedByte, &ack, 1 - sequenceNum);
454
455 // Check if the acknowledgment is as expected
456 if (state == DONE && ack == ACK(1 - sequenceNum))
457 {
458 sequenceNum = 1 - sequenceNum;
459 alarm(0); // Cancel the alarm
460 STOP = TRUE; // Stop the loop
461 printf("RECEIVED ACK aka RR...\n");
462 }
463 // Resend the frame if NACK received
464 else if (state == DONE)
465 {
466 printf("RECEIVED NACK aka RREJ...\n");
467 write(fd, message, size);
468 state = START;
469 }
470 }
471 }
472 return 0; // Return success
473}

479// Function to process received data and handle byte stuffing return true when it must return ack, and false for nack
480int receiveData(unsigned char *packet, int sequenceNum, size_t *size_read)
481{
482 state = START;
483 unsigned char REPLY_C = (1 - sequenceNum) << 7;
484 unsigned char CONTROL_C = sequenceNum << 7; // Construct control byte using sequence number
485 unsigned int BCC2 = 0, i = 0, stuffing = 0; // Stuffing flag: set to 1 every time an ESC is encountered
486
487 // Keep processing bytes until the end flag is received
488 while (state != DONE)
489 {
490 unsigned char receivedByte;
491 unsigned int bytesNum = read(fd, &receivedByte, 1);
492
493 if (bytesNum > 0)
494 {
495
496 switch (state)
497 {
498 case START:
499 if (receivedByte == FLAG)
500 {
501 state = FLAG_RCV;
502 }
503 break;
504
505 case A_RCV:
506 if (receivedByte == FLAG)
507 {
508 state = FLAG_RCV;
509 }
510 else if (receivedByte == CONTROL_C)
511 {
512 state = C_RCV;
513 }
514 // Handling case of receiving repeated message and waiting for the next one
515 else if (REPLY_C == receivedByte)
516 {
517 return REPEATED_MSG_CODE; // Return code indicating a repeated message was received
518 }
519 else
520 {
521 state = START;
522 }
523 break;
524
525 case FLAG_RCV:
526 if (receivedByte == FLAG)
527 {
528 state = FLAG_RCV;
529 }
530 else if (receivedByte == A)
531 {
532 state = A_RCV;
533 }
534 else
535 {
536 state = START;
537 }
538 break;
539
540 case C_RCV:
541 if (receivedByte == FLAG)
542 {
543 state = FLAG_RCV;
544 }
545 else if (receivedByte == BCC(A, CONTROL_C))
546 {
547 state = BCC_DATA;
548 }
549 else
550 {
551 state = START;
552 }
553 break;
554
555 // Handle byte stuffing and frame de-stuffing
556 case BCC_DATA:
557 if (!stuffing)
558 {
559 if (receivedByte == FLAG)
560 {
561 state = DONE;
562 BCC2 = BCC(BCC2, packet[i - 1]);
563 *size_read = i - 1; // Update the size of the read data
564 return (BCC2 == packet[i - 1]); // Return TRUE if BCC matches, otherwise FALSE
565 }
566 else if (receivedByte == ESC)
567 {
568 stuffing = TRUE; // Set the stuffing flag if ESC is encountered
569 }
570 else
571 {
572 BCC2 = BCC(BCC2, receivedByte);
573 packet[i] = receivedByte;
574 i++;
575 }
576 }
577 else
578 {
579 stuffing = FALSE; // Reset the stuffing flag
580 BCC2 = BCC(BCC2, receivedByte ^ 0x20);
581 packet[i] = receivedByte ^ 0x20;
582 i++;
583 }
584 break;
585
586 default:
587 break;
588 }
589 }
590 }
591
592 return FALSE;
593}

597int llread(unsigned char *packet)
598{

601
602
603 // Continuously try to receive data until successful

605 {
606 // If the reply indicates an error (e.g., checksum mismatch)
607 if (readStatus == 0)
608 {
609 printf("Sending RRej or NACK\n");
610

612 unsigned char NACK_C = NACK(1 - sequenceNum);
613 unsigned char buf[] = {FLAG, A, NACK_C, BCC(A, NACK_C), F};
614 write(fd, buf, 5);
615 }
616 // If the received message is a duplicate (e.g., retransmission)
617 else
618 {
619 printf("Repeated message, sending ACK\n");
620
621 // Send an ACK to prevent further retransmissions
622 unsigned char ACK_C = ACK(sequenceNum);
623 unsigned char buf[] = {FLAG, A, ACK_C, BCC(A, ACK_C), F};
624 write(fd, buf, 5);
625 }
626 }
627
628 // Once data is received correctly, send an ACK
629 printf("Everything in order, sending ACK\n");
630 sequenceNum = 1 - sequenceNum; // Toggle the sequence number
631 unsigned char ACK_C = ACK(sequenceNum);
632 unsigned char buf[] = {FLAG, A, ACK_C, BCC(A, ACK_C), F};
633 write(fd, buf, 5);
